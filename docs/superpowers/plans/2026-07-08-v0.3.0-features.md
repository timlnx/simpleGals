# simpleGals v0.3.0 Features Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add three gated, user-facing features to simpleGals (social-media preview images/tags, a camera-metadata EXIF block, and a downloadable gallery zip), plus a single-source versioning refactor.

**Architecture:** Extend the existing core library (config -> processor -> workers -> gallery build -> Jinja templates) without changing its shape. New per-image artifacts (`_og.jpg`) and cached EXIF ride the existing sidecar/staleness machinery; the zip is a build-time artifact with its own incremental manifest; three new `ProjectConfig` booleans gate everything and surface in `sgui`.

**Tech Stack:** Python >= 3.10, Pillow, piexif, bitmath >= 2.0.0, Jinja2, urwid, hatchling, pytest.

**Companion spec:** `docs/superpowers/specs/2026-07-08-v0.3.0-features-design.md` (the approved PRD). Read it for full rationale; this plan is the executable version. Where CSS/markup is long, the spec holds the canonical copy and this plan reproduces the essential parts.

## Global Constraints

- Python floor: `>=3.10`. No new runtime dependencies (Pillow, piexif, bitmath, jinja2, urwid already present).
- File sizes use bitmath with NIST prefixes (MiB, not MB). Zip button size: `bitmath.Byte(n).best_prefix().format("{value:.0f} {unit}")` (integer, no decimals).
- No em-dashes in any prose, comments, or copy. Use plain punctuation.
- No Claude co-author attribution in commits.
- Config keys are snake_case (`social_previews`, `exif_display`, `gallery_zip`).
- Version strings are bare PEP 440 everywhere (no `v` prefix): `VERSION` file, `pyproject.toml`, git tag, GitHub Release name.
- Run the full suite with `make test` (or `pytest tests/`). The suite is currently green (145 tests); keep it green after every task.

```python
# simplegals/core/exif.py
from __future__ import annotations

from fractions import Fraction
from pathlib import Path

from PIL import Image
from PIL.ExifTags import Base, IFD


def _shutter(value) -> str:
    try:
        f = Fraction(value).limit_denominator(8000)
    except (TypeError, ValueError):
        return str(value)
    if f >= 1:
        return f"{float(f):g}s"
    return f"{f.numerator}/{f.denominator}s"


def _clean(value) -> str:
    return str(value).strip().strip("\x00").strip()


def extract_exif(source: Path) -> dict | None:
    """Return a dict of display-ready camera-metadata strings, or None if absent.

    Only keys with usable values are included, so callers can render just the
    fields that exist.
    """
    try:
        with Image.open(source) as img:
            exif = img.getexif()
    except Exception:
        return None
    if not exif:
        return None

    ifd = exif.get_ifd(IFD.Exif)
    out: dict = {}

    make = _clean(exif.get(Base.Make.value, ""))
    model = _clean(exif.get(Base.Model.value, ""))
    camera = " ".join(p for p in (make, model) if p).strip()
    # Avoid duplicating make when model already contains it (e.g. "NIKON Z 7")
    if make and model and model.upper().startswith(make.split()[0].upper()):
        camera = model
    if camera:
        out["camera"] = camera

    lens = _clean(ifd.get(Base.LensModel.value, "")) or _clean(exif.get(Base.LensModel.value, ""))
    if lens:
        out["lens"] = lens

    fnum = ifd.get(Base.FNumber.value)
    iso = ifd.get(Base.ISOSpeedRatings.value) or exif.get(Base.ISOSpeedRatings.value)
    shutter = ifd.get(Base.ExposureTime.value)
    parts = []
    if fnum:
        parts.append(f"f/{float(fnum):g}")
    if iso:
        parts.append(f"ISO {int(iso)}")
    if shutter:
        parts.append(_shutter(shutter))
    if parts:
        out["exposure"] = " · ".join(parts)

    focal = ifd.get(Base.FocalLength.value)
    if focal:
        out["focal_length"] = f"{float(focal):g}mm"

    flash = ifd.get(Base.Flash.value)
    if flash is not None:
        out["flash"] = "Fired" if int(flash) & 1 else "Did not fire"

    bias = ifd.get(Base.ExposureBiasValue.value)
    if bias is not None and float(bias) != 0.0:
        out["exposure_comp"] = f"{float(bias):+g} EV"

    wb = ifd.get(Base.WhiteBalance.value)
    if wb is not None:
        out["white_balance"] = {0: "Auto", 1: "Manual"}.get(int(wb), str(wb))

    metering = ifd.get(Base.MeteringMode.value)
    if metering is not None:
        out["metering"] = {
            0: "Unknown", 1: "Average", 2: "Center-weighted", 3: "Spot",
            4: "Multi-spot", 5: "Pattern", 6: "Partial",
        }.get(int(metering), str(metering))

    return out or None
```

```python
# simplegals/core/archive.py
from __future__ import annotations

import hashlib
import json
import re
import zipfile
from pathlib import Path

from .metadata import file_sha256

_STATE_FILE = "gallery-zip.json"


def gallery_zip_name(title: str) -> str:
    slug = re.sub(r"[^\w.-]+", "_", (title or "").strip()).strip("_")
    return f"{slug or 'gallery'}.zip"


def compute_manifest(out_dir: Path, names: list[str]) -> str:
    h = hashlib.sha256()
    for name in sorted(names):
        p = out_dir / name
        sha = file_sha256(p) if p.exists() else "missing"
        h.update(f"{name}:{sha}\n".encode())
    return h.hexdigest()[:16]


def build_zip(out_dir: Path, names: list[str], zip_path: Path, progress_cb=None) -> tuple[int, int]:
    ordered = sorted(names)
    total = len(ordered)
    tmp = zip_path.with_suffix(".zip.tmp")
    count = 0
    with zipfile.ZipFile(tmp, "w", compression=zipfile.ZIP_STORED) as z:
        for name in ordered:
            src = out_dir / name
            if not src.exists():
                continue
            z.write(src, arcname=name)
            count += 1
            if progress_cb:
                progress_cb(count, total)
    tmp.replace(zip_path)
    return count, zip_path.stat().st_size


def load_zip_state(meta_dir: Path) -> dict | None:
    p = meta_dir / _STATE_FILE
    if not p.exists():
        return None
    try:
        return json.loads(p.read_text(encoding="utf-8"))
    except (ValueError, OSError):
        return None


def save_zip_state(meta_dir: Path, state: dict) -> None:
    meta_dir.mkdir(parents=True, exist_ok=True)
    (meta_dir / _STATE_FILE).write_text(json.dumps(state, indent=2), encoding="utf-8")
```
